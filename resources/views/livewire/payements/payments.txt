        function closeModal(modalId) {
            document.getElementById(modalId).classList.add('hidden');
            // Also call Livewire to update the state
            if (modalId === 'create-payement-modal') {
                Livewire.first().set('showCreateModal', false);
            } else if (modalId === 'edit-payement-modal') {
                Livewire.first().set('showEditModal', false);
            } else if (modalId === 'delete-payement-modal') {
                Livewire.first().set('showDeleteModal', false);
            }
        }

        // Listen for Livewire events to show/hide modals
        document.addEventListener('livewire:init', () => {
            Livewire.on('showCreateModal', () => {
                document.getElementById('create-payement-modal').classList.remove('hidden');
            });

            Livewire.on('showEditModal', () => {
                document.getElementById('edit-payement-modal').classList.remove('hidden');
            });

            Livewire.on('showDeleteModal', () => {
                document.getElementById('delete-payement-modal').classList.remove('hidden');
